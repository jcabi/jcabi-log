51  /**
52   * Utility class for getting {@code stdout} from a running process
53   * and logging it through SLF4J.
54   *
55   * <p>For example:
56   *
57   * <pre> String name = new VerboseProcess(
58   *   new ProcessBuilder("who", "am", "i")
59   * ).stdout();</pre>
60   *
61   * <p>The class throws an exception if the process returns a non-zero exit
62   * code.
63   *
64   * <p>The class is thread-safe.
65   *
66   * @author Yegor Bugayenko (yegor@tpc2.com)
67   * @version $Id$
68   * @since 0.5
69   */
70  @ToString
71  @EqualsAndHashCode(of = "process")
72  public final class VerboseProcess {
73  
74      /**
75       * The process we're working with.
76       */
77      private final transient Process process;
78  
79      /**
80       * Log level for stdout.
81       */
82      private final transient Level olevel;
83  
84      /**
85       * Log level for stderr.
86       */
87      private final transient Level elevel;
88  
89      /**
90       * Public ctor.
91       * @param prc The process to work with
92       */
93      public VerboseProcess(final Process prc) {
94          this(prc, Level.INFO, Level.WARNING);
95      }
96  
97      /**
98       * Public ctor (builder will be configured to redirect error input to
99       * the {@code stdout} and will receive an empty {@code stdin}).
100      * @param builder Process builder to work with
101      */
102     public VerboseProcess(
103         @NotNull(message = "process builder can't be NULL")
104         final ProcessBuilder builder) {
105         this(VerboseProcess.start(builder));
106     }

108     /**
109      * Public ctor, with a given process and logging levels for {@code stdout}
110      * and {@code stderr}.
111      * @param prc Process to execute and monitor
112      * @param stdout Log level for stdout
113      * @param stderr Log level for stderr
114      * @since 0.11
115      */
116     public VerboseProcess(
117         @NotNull(message = "process can't be NULL") final Process prc,
118         @NotNull(message = "stdout level can't be NULL") final Level stdout,
119         @NotNull(message = "stderr level can't be NULL") final Level stderr) {
120         this.process = prc;
121         this.olevel = stdout;
122         this.elevel = stderr;
123     }

125     /**
126      * Public ctor, with a given process and logging levels for {@code stdout}

128      * @param bdr Process builder to execute and monitor
129      * @param stdout Log level for stdout
130      * @param stderr Log level for stderr

133     public VerboseProcess(
134         @NotNull(message = "process can't be NULL") final ProcessBuilder bdr,
135         @NotNull(message = "stdout level can't be NULL") final Level stdout,
136         @NotNull(message = "stderr level can't be NULL") final Level stderr) {
137         this(VerboseProcess.start(bdr), stdout, stderr);
138     }

140     /**
141      * Get {@code stdout} from the process, after its finish (the method will
142      * wait for the process and log its output).
143      *
144      * <p>The method will check process exit code, and if it won't be equal
145      * to zero a runtime exception will be thrown. A non-zero exit code
146      * usually is an indicator of problem. If you want to ignore this code,
147      * use {@link #stdoutQuietly()} instead.
148      *
149      * @return Full {@code stdout} of the process
150      */
151     public String stdout() {
152         return this.stdout(true);
153     }
154 
155     /**
156      * Get {@code stdout} from the process, after its finish (the method will
157      * wait for the process and log its output).
158      *
159      * <p>This method ignores exit code of the process. Even if it is
160      * not equal to zero (which usually is an indicator of an error), the
161      * method will quietly return its output. The method is useful when
162      * you're running a background process. You will kill it with
163      * {@link Process#destroy()}, which usually will lead to a non-zero
164      * exit code, which you want to ignore.
165      *
166      * @return Full {@code stdout} of the process
167      * @since 0.10
168      */
169     public String stdoutQuietly() {
170         return this.stdout(false);
171     }

173     /**
174      * Start a process from the given builder.
175      * @param builder Process builder to work with
176      * @return Process started
177      */
178     private static Process start(@NotNull final ProcessBuilder builder) {
179         builder.redirectErrorStream(true);
180         try {
181             final Process process = builder.start();
182             process.getOutputStream().close();
183             return process;
184         } catch (final IOException ex) {
185             throw new IllegalStateException(ex);
186         }
187     }
188 
189     /**
190      * Get standard output and check for non-zero exit code (if required).
191      * @param check TRUE if we should check for non-zero exit code
192      * @return Full {@code stdout} of the process
193      */
194     private String stdout(final boolean check) {
195         final long start = System.currentTimeMillis();
196         final String stdout;
197         try {
198             stdout = this.waitFor();
199         } catch (final InterruptedException ex) {
200             Thread.currentThread().interrupt();
201             throw new IllegalStateException(ex);
202         }
203         final int code = this.process.exitValue();
204         Logger.debug(
205             this,
206             "#stdout(): process %s completed (code=%d, size=%d) in %[ms]s",
207             this.process, code, stdout.length(),
208             System.currentTimeMillis() - start
209         );
210         if (check && code != 0) {
211             throw new IllegalArgumentException(
212                 Logger.format("Non-zero exit code %d: %[text]s", code, stdout)
213             );
214         }
215         return stdout;
216     }

218     /**
219      * Wait for the process to stop, logging its output in parallel.
220      * @return Stdout produced by the process
221      * @throws InterruptedException If interrupted in between
222      */
223     private String waitFor() throws InterruptedException {
224         final CountDownLatch done = new CountDownLatch(2);
225         final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
226         Logger.debug(
227             this,
228             "#waitFor(): waiting for stdout of %s in %s...",
229             this.process,
230             this.monitor(
231                 this.process.getInputStream(),
232                 done, stdout, this.olevel
233             )
234         );
235         Logger.debug(
236             this,
237             "#waitFor(): waiting for stderr of %s in %s...",
238             this.process,
239             this.monitor(
240                 this.process.getErrorStream(),
241                 done, new ByteArrayOutputStream(), this.elevel
242             )
243         );
244         try {
245             this.process.waitFor();
246         } finally {
247             Logger.debug(
248                 this, "#waitFor(): process finished : %s", this.process
249             );
250             if (!done.await(2L, TimeUnit.SECONDS)) {
251                 Logger.error(this, "#wait() failed");
252             }
253         }
254         try {
255             return stdout.toString("UTF-8");
256         } catch (UnsupportedEncodingException ex) {
257             throw new IllegalStateException(ex);
258         }
259     }

261     /**
262      * Monitor this input input.
263      * @param input Stream to monitor
264      * @param done Count down latch to signal when done
265      * @param output Buffer to write to
266      * @param level Logging level
267      * @return Thread which is monitoring
268      * @checkstyle ParameterNumber (5 lines)
269      */
270     @SuppressWarnings("PMD.DoNotUseThreads")
271     private Thread monitor(final InputStream input, final CountDownLatch done,
272         final OutputStream output, final Level level) {
273         final Thread thread = new Thread(
274             new VerboseRunnable(
275                 new VerboseProcess.Monitor(input, done, output, level),
276                 false
277             )
278         );
279         thread.setName("VerboseProcess");
280         thread.setDaemon(true);
281         thread.start();
282         return thread;
283     }

285     /**
286      * Stream monitor.
287      */
288     private static final class Monitor implements Callable<Void> {
289         /**
290          * Stream to read.
291          */
292         private final transient InputStream input;
293         /**
294          * Latch to count down when done.
295          */
296         private final transient CountDownLatch done;
297         /**
298          * Buffer to save output.
299          */
300         private final transient OutputStream output;
301         /**
302          * Log level.
303          */
304         private final transient Level level;
305         /**
306          * Ctor.
307          * @param inp Stream to monitor
308          * @param latch Count down latch to signal when done
309          * @param out Buffer to write to
310          * @param lvl Logging level
311          * @checkstyle ParameterNumber (5 lines)
312          */
313         Monitor(final InputStream inp, final CountDownLatch latch,
314             final OutputStream out, final Level lvl) {
315             this.input = inp;
316             this.done = latch;
317             this.output = out;
318             this.level = lvl;
319         }
320         @Override
321         public Void call() throws Exception {
322             final BufferedReader reader = new BufferedReader(
323                 new InputStreamReader(
324                     this.input,
325                     Charset.forName(CharEncoding.UTF_8)
326                 )
327             );
328             final BufferedWriter writer = new BufferedWriter(
329                 new OutputStreamWriter(
330                     this.output,
331                     Charset.forName(CharEncoding.UTF_8)
332                 )
333             );
334             try {
335                 while (true) {
336                     final String line = reader.readLine();
337                     if (line == null) {
338                         break;
339                     }
340                     Logger.log(
341                         this.level, VerboseProcess.class,
342                         ">> %s", line
343                     );
344                     writer.write(line);
345                     writer.newLine();
346                 }
347                 this.done.countDown();
348             } finally {
349                 try {
350                     reader.close();
351                     writer.close();
352                 } catch (final IOException ex) {
353                     Logger.error(
354                         this,
355                         "failed to close reader: %[exception]s", ex
356                     );
357                 }
358             }
359             return null;
360         }
361     }
