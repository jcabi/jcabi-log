40  /**
41   * Universal logger, and adapter between your app and SLF4J API.
42   *
43   * <p>Instead of relying
44   * on some logging engine you can use this class, which transforms all
45   * messages to SLF4J. This approach gives you a perfect decoupling of business
46   * logic and logging mechanism. All methods in the class are called
47   * statically, without the necessity to instantiate the class.
48   *
49   * <p>Use it like this in any class, and in any package:
50   *
51   * <pre> package com.example.XXX;
52   * import com.jcabi.log.Logger;
53   * public class MyClass {
54   *   public void foo(Integer num) {
55   *     Logger.info(this, "foo(%d) just called", num);
56   *   }
57   * }</pre>
58   *
59   * <p>Or statically (pay attention to {@code MyClass.class}):
60   *
61   * <pre> public class MyClass {
62   *   public static void foo(Integer num) {
63   *     Logger.info(MyClass.class, "foo(%d) just called", num);
64   *   }
65   * }</pre>
66   *
67   * <p>Exact binding between SLF4J and logging facility has to be
68   * specified in {@code pom.xml} of your project (or in classpath directly).
69   *
70   * <p>For performance reasons in most cases before sending a
71   * {@code TRACE} or {@code DEBUG} log message you may check whether this
72   * logging level is enabled in the project, e.g.:
73   *
74   * <pre> //...
75   * if (Logger.isTraceEnabled(this)) {
76   *   Logger.trace(this, "#foo() called");
77   * }
78   * //...</pre>
79   *
80   * <p>There is only one reason to do so - if you want to save time spent on
81   * preparing of the arguments. By default, such a call is made inside every
82   * method of {@link Logger} class.
83   *
84   * @since 0.1
85   */

110     /**
111      * Format one string.
112      * @param fmt The format
113      * @param args List of arbitrary arguments
114      * @return Formatted string
115      */
116     public static String format(final String fmt, final Object... args) {
117         final String result;
118         if (args.length == 0) {
119             result = fmt;
120         } else {
121             final PreFormatter pre = new PreFormatter(fmt, args);
122             result = String.format(pre.getFormat(), pre.getArguments());
123         }
124         return result;
125     }

346     /**
347      * Is the given logging level enabled?
348      * @param level The level of logging
349      * @param source The source of the logging operation
350      * @return Is it enabled?
351      * @since 0.13
352      */
353     public static boolean isEnabled(final Level level, final Object source) {
354         boolean enabled = false;
355         if (level.equals(Level.SEVERE)) {
356             enabled = Logger.logger(source).isErrorEnabled();
357         } else if (level.equals(Level.WARNING)) {
358             enabled = Logger.logger(source).isWarnEnabled();
359         } else if (level.equals(Level.INFO) || level.equals(Level.CONFIG)) {
360             enabled = Logger.logger(source).isInfoEnabled();
361         } else if (level.equals(Level.FINE) || level.equals(Level.ALL)) {
362             enabled = Logger.logger(source).isDebugEnabled();
363         } else if (level.equals(Level.FINER) || level.equals(Level.FINEST)) {
364             enabled = Logger.logger(source).isTraceEnabled();
365         }
366         return enabled;
367     }

369     /**
370      * Log one line using the logging level specified.
371      * @param level The level of logging
372      * @param source The source of the logging operation
373      * @param msg The text message to be logged
374      * @param args Optional arguments for string formatting
375      * @since 0.8
376      * @checkstyle ParameterNumber (4 lines)
377      */
378     public static void log(final Level level, final Object source,
379         final String msg, final Object... args) {
380         if (level.equals(Level.SEVERE)) {
381             Logger.error(source, msg, args);
382         } else if (level.equals(Level.WARNING)) {
383             Logger.warn(source, msg, args);
384         } else if (level.equals(Level.INFO) || level.equals(Level.CONFIG)) {
385             Logger.info(source, msg, args);
386         } else if (level.equals(Level.FINE) || level.equals(Level.ALL)) {
387             Logger.debug(source, msg, args);
388         } else if (level.equals(Level.FINER) || level.equals(Level.FINEST)) {
389             Logger.trace(source, msg, args);
390         }
391     }

393     /**
394      * Returns an {@link OutputStream}, which converts all incoming data
395      * into logging lines (separated by {@code \x0A} in UTF-8).
396      * @param level The level of logging
397      * @param source The source of the logging operation
398      * @return Output stream directly pointed to the logging facility
399      * @see <a href="http://stackoverflow.com/questions/17258325">some discussion</a>
400      * @since 0.8
401      * @checkstyle MagicNumberCheck (20 lines)
402      */
403     public static OutputStream stream(final Level level, final Object source) {
404         // @checkstyle AnonInnerLengthCheck (50 lines)
405         return new OutputStream() {
406             private final transient ByteArrayOutputStream buffer =
407                 new ByteArrayOutputStream();
408             @Override
409             public void write(final int data) throws IOException {
410                 if (data == '\n') {
411                     Logger.log(
412                         level, source,
413                         this.buffer.toString(Logger.UTF_8)
414                     );
415                     this.buffer.reset();
416                 } else if (data >= 0x20 && data <= 0x7f) {
417                     this.buffer.write(data);
418                 } else {
419                     this.buffer.write(
420                         String.format(
421                             "\\x%02x", data & 0xff
422                         ).getBytes(Logger.UTF_8)
423                     );
424                 }
425             }
426         };
427     }

429     /**
430      * Log messages constructed from Suppliers.
431      * It is more efficient to use method referencing because the method
432      * won't be called unless the specified logging level is enabled.
433      *
434      * This saves you the effort of having to check if the level is enabled
435      * before calling the logging method.
436      * E.g.
437      * <pre>
438      *     if (Logger.isDebugEnabled(this)) {
439      *         Logger.debug(this, "Some %s", calculate());
440      *     }
441      * </pre><br>
442      * turns into <br>
443      * <pre>
444      *     Logger.withSupplier().debug(this, "Some %s", this::calculate());
445      * </pre><br>
446      * and the calculate() method won't be called unless the debug level is
447      * active.
448      *
449      * @return Object containing methods for logging with Supplier-constructed
450      *  messages
451      */
452     public static SupplierLogger withSupplier() {
453         return new SupplierLogger();
454     }

456     /**
457      * Set final static field in order to fix the %L log4j parameter.
458      * @param field Field
459      * @param value New value
460      * @throws NoSuchFieldException If some problem
461      * @throws IllegalAccessException If some problem
462      * @checkstyle ThrowsCountCheck (4 lines)
463      */
464     private static void setFinalStatic(final Field field, final Object value)
465         throws NoSuchFieldException, IllegalAccessException {
466         field.setAccessible(true);
467         final Field modifiers = Field.class.getDeclaredField("modifiers");
468         modifiers.setAccessible(true);
469         modifiers.setInt(field, field.getModifiers() & ~Modifier.FINAL);
470         field.set(null, value);
471     }

473     /**
474      * Get the instance of the logger for this particular caller.
475      * @param source Source of the logging operation
476      * @return The instance of {@code Logger} class
477      */
478     private static org.slf4j.Logger logger(final Object source) {
479         final org.slf4j.Logger logger;
480         if (source instanceof Class) {
481             logger = LoggerFactory.getLogger((Class<?>) source);
482         } else if (source instanceof String) {
483             logger = LoggerFactory.getLogger(String.class.cast(source));
484         } else {
485             logger = LoggerFactory.getLogger(source.getClass());
486         }
487         if ("org.slf4j.impl.Log4jLoggerAdapter"
488             .equals(logger.getClass().getName())) {
489             try {
490                 final Field fqcn = logger.getClass()
491                     .getDeclaredField("FQCN");
492                 setFinalStatic(fqcn, Logger.class.getName());
493             } catch (final NoSuchFieldException | IllegalAccessException ex) {
494                 throw new IllegalStateException(ex);
495             }
496         }
497         return logger;
498     }
