30  package com.jcabi.log;
31  
32  import java.util.Map;
33  import java.util.concurrent.ConcurrentHashMap;
34  import java.util.concurrent.ConcurrentMap;
35  import org.apache.log4j.EnhancedPatternLayout;
36  import org.apache.log4j.Level;
37  import org.apache.log4j.spi.LoggingEvent;
38  
39  /**
40   * Multi-color layout for LOG4J.
41   *
42   * <p>Use it in your LOG4J configuration:
43   *
44   * <pre> log4j.rootLogger=INFO, CONSOLE
45   * log4j.appender.CONSOLE=org.apache.log4j.ConsoleAppender
46   * log4j.appender.CONSOLE.layout=com.jcabi.log.MulticolorLayout
47   * log4j.appender.CONSOLE.layout.ConversionPattern=[%color{%-5p}] %c: %m%n</pre>
48   *
49   * <p>The part of the message wrapped with {@code %color{...}}
50   * will change its color according to the logging level of the event. Without
51   * this highlighting the behavior of the layout is identical to
52   * {@link EnhancedPatternLayout}. You can use {@code %color-red{...}} if you
53   * want to use specifically red color for the wrapped piece of text. Supported
54   * colors are: {@code red}, {@code blue}, {@code yellow}, {@code cyan},
55   * {@code black}, and {@code white}.
56   *
57   * <p>Besides that you can specify any ANSI color you like with
58   * {@code %color-<attr>;<bg>;<fg>{...}}, where
59   * {@code <attr>} is a binary mask of attributes,
60   * {@code <bg>} is a background color, and
61   * {@code <fg>} is a foreground color. Read more about
62   * <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>.
63   *
64   * <p>This class or its parents are <b>not</b> serializable.
65   *
66   * <p>Maven dependency for this class is
67   * (see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How
68   * to use with Maven</a> instructions):
69   *
70   * <pre>&lt;dependency&gt;
71   *  &lt;groupId&gt;com.jcabi&lt;/groupId&gt;
72   *  &lt;artifactId&gt;jcabi-log&lt;/artifactId&gt;
73   * &lt;/dependency&gt;</pre>
74   * @since 0.1.10
75   * @see <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>
76   * @see <a href="http://logging.apache.org/log4j/1.2/apidocs/org/apache/log4j/PatternLayout.html">PatternLayout from LOG4J</a>
77   * @see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How to use with Maven</a>
78   */
79  @SuppressWarnings("PMD.NonStaticInitializer")
80  public final class MulticolorLayout extends EnhancedPatternLayout {
81  
82      /**
83       * Name of the property that is used to disable log coloring.
84       */
85      private static final String COLORING_PROPERY = "com.jcabi.log.coloring";
86  
87      /**
88       * Colors of levels.
89       */
90      private final transient ConcurrentMap<String, String> levels =
91          MulticolorLayout.levelMap();
92  
93      /**
94       * Store original conversation pattern to be able
95       * to recalculate it, if new colors are provided.
96       */
97      private transient String base;
98  
99      /**
100      * Color human readable data.
101      */
102     private final transient Colors colors = new Colors();
103 
104     @Override
105     public void setConversionPattern(final String pattern) {
106         this.base = pattern;
107         super.setConversionPattern(
108             new ConversionPattern(this.base, this.colors).generate()
109         );
110     }
111 
112     /**
113      * Allow to overwrite or specify new ANSI color names
114      * in a javascript map like format.
115      *
116      * @param cols JavaScript like map of color names
117      * @since 0.9
118      */
119     @SuppressWarnings("PMD.UseConcurrentHashMap")
120     public void setColors(final String cols) {
121         final Map<String, String> parsed = new ParseableInformation(
122             cols
123         ).information();
124         for (final Map.Entry<String, String> entry : parsed.entrySet()) {
125             this.colors.addColor(entry.getKey(), entry.getValue());
126         }
127         if (this.base != null) {
128             this.setConversionPattern(this.base);
129         }
130     }
131 
132     /**
133      * Allow to overwrite the ANSI color values for the log levels
134      * in a javascript map like format.
135      * @param lev JavaScript like map of levels
136      * @since 0.9
137      */
138     @SuppressWarnings("PMD.UseConcurrentHashMap")
139     public void setLevels(final String lev) {
140         final Map<String, String> parsed = new ParseableLevelInformation(
141             lev
142         ).information();
143         for (final Map.Entry<String, String> entry : parsed.entrySet()) {
144             this.levels.put(entry.getKey(), entry.getValue());
145         }
146     }
147 
148     @Override
149     public String format(final LoggingEvent event) {
150         final Formatted formatted;
151         if (MulticolorLayout.isColoringEnabled()) {
152             formatted = this.colorfulFormatting(event);
153         } else {
154             formatted = this.dullFormatting(event);
155         }
156         return formatted.format();
157     }
158 
159     /**
160      * Generate a dull {@code Formatted}.
161      * @param event Event to be formatted
162      * @return A {@link Formatted} to format the event
163      * @checkstyle NonStaticMethodCheck (10 lines)
164      */
165     private Formatted dullFormatting(final LoggingEvent event) {
166         return new DullyFormatted(super.format(event));
167     }
168 
169     /**
170      * Generate a colorful {@code Formatted}.
171      * @param event Event to be formatted
172      * @return Text of a log event, probably colored with ANSI color codes
173      */
174     private Formatted colorfulFormatting(final LoggingEvent event) {
175         return new ColorfullyFormatted(
176             super.format(event),
177             this.levels.get(event.getLevel().toString())
178         );
179     }
180 
181     /**
182      * Level map.
183      * @return Map of levels
184      */
185     private static ConcurrentMap<String, String> levelMap() {
186         final ConcurrentMap<String, String> map = new ConcurrentHashMap<>(0);
187         map.put(Level.TRACE.toString(), "2;33");
188         map.put(Level.DEBUG.toString(), "2;37");
189         map.put(Level.INFO.toString(), "0;37");
190         map.put(Level.WARN.toString(), "0;33");
191         map.put(Level.ERROR.toString(), "0;31");
192         map.put(Level.FATAL.toString(), "0;35");
193         return map;
194     }
195 
196     /**
197      * Should the logged text be colored or not.
198      * @return True if the coloring is enabled, or false otherwise.
199      */
200     private static boolean isColoringEnabled() {
201         return !"false".equals(
202             System.getProperty(MulticolorLayout.COLORING_PROPERY)
203         );
204     }
205 
206 }
