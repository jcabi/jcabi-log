30  package com.jcabi.log;
31  
32  import java.util.Locale;
33  import java.util.concurrent.ConcurrentHashMap;
34  import java.util.concurrent.ConcurrentMap;
35  import java.util.regex.Matcher;
36  import java.util.regex.Pattern;
37  import lombok.EqualsAndHashCode;
38  import lombok.ToString;
39  import org.apache.log4j.EnhancedPatternLayout;
40  import org.apache.log4j.Level;
41  import org.apache.log4j.spi.LoggingEvent;
42  
43  /**
44   * Multi-color layout for LOG4J.
45   *
46   * <p>Use it in your LOG4J configuration:
47   *
48   * <pre> log4j.rootLogger=INFO, CONSOLE
49   * log4j.appender.CONSOLE=org.apache.log4j.ConsoleAppender
50   * log4j.appender.CONSOLE.layout=com.jcabi.log.MulticolorLayout
51   * log4j.appender.CONSOLE.layout.ConversionPattern=[%color{%-5p}] %c: %m%n</pre>
52   *
53   * <p>The part of the message wrapped with {@code %color{...}}
54   * will change its color according to the logging level of the event. Without
55   * this highlighting the behavior of the layout is identical to
56   * {@link EnhancedPatternLayout}. You can use {@code %color-red{...}} if you
57   * want to use specifically red color for the wrapped piece of text. Supported
58   * colors are: {@code red}, {@code blue}, {@code yellow}, {@code cyan},
59   * {@code black}, and {@code white}.
60   *
61   * <p>Besides that you can specify any ANSI color you like with
62   * {@code %color-<attr>;<bg>;<fg>{...}}, where
63   * {@code <attr>} is a binary mask of attributes,
64   * {@code <bg>} is a background color, and
65   * {@code <fg>} is a foreground color. Read more about
66   * <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>.
67   *
68   * <p>This class or its parents are <b>not</b> serializable.
69   *
70   * <p>Maven dependency for this class is
71   * (see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How
72   * to use with Maven</a> instructions):
73   *
74   * <pre>&lt;dependency&gt;
75   *  &lt;groupId&gt;com.jcabi&lt;/groupId&gt;
76   *  &lt;artifactId&gt;jcabi-log&lt;/artifactId&gt;
77   * &lt;/dependency&gt;</pre>
78   *
79   * @author Yegor Bugayenko (yegor@tpc2.com)
80   * @version $Id$
81   * @since 0.1.10
82   * @see <a href="http://en.wikipedia.org/wiki/ANSI_escape_code">ANSI escape code</a>
83   * @see <a href="http://logging.apache.org/log4j/1.2/apidocs/org/apache/log4j/PatternLayout.html">PatternLayout from LOG4J</a>
84   * @see <a href="http://www.jcabi.com/jcabi-log/multicolor.html">How to use with Maven</a>
85   */
86  @ToString
87  @EqualsAndHashCode(callSuper = false)
88  @SuppressWarnings("PMD.NonStaticInitializer")
89  public final class MulticolorLayout extends EnhancedPatternLayout {
90  
91      /**
92       * Control sequence indicator.
93       */
94      private static final String CSI = "\u001b[";
95  
96      /**
97       * To split strings with javascript like map syntax.
98       */
99      private static final String SPLIT_ITEMS = ",";
100 
101     /**
102      * To split key:value pairs.
103      */
104     private static final String SPLIT_VALUES = ":";
105 
106     /**
107      * Regular expression for all matches.
108      */
109     private static final Pattern METAS = Pattern.compile(
110         "%color(?:-([a-z]+|[0-9]{1,3};[0-9]{1,3};[0-9]{1,3}))?\\{(.*?)\\}"
111     );
112 
113     /**
114      * Colors with names.
115      */
116     private final transient ConcurrentMap<String, String> colors =
117         MulticolorLayout.colorMap();
118 
119     /**
120      * Colors of levels.
121      */
122     private final transient ConcurrentMap<String, String> levels =
123         MulticolorLayout.levelMap();
124 
125     /**
126      * Store original conversation pattern to be able
127      * to recalculate it, if new colors are provided.
128      */
129     private transient String base;
130 
131     @Override
132     public void setConversionPattern(final String pattern) {
133         this.base = pattern;
134         final Matcher matcher = MulticolorLayout.METAS.matcher(pattern);
135         final StringBuffer buf = new StringBuffer(0);
136         while (matcher.find()) {
137             matcher.appendReplacement(buf, "");
138             buf.append(MulticolorLayout.CSI)
139                 .append(this.ansi(matcher.group(1)))
140                 .append('m')
141                 .append(matcher.group(2))
142                 .append(MulticolorLayout.CSI)
143                 .append('m');
144         }
145         matcher.appendTail(buf);
146         super.setConversionPattern(buf.toString());
147     }
148 
149     /**
150      * Allow to overwrite or specify new ANSI color names
151      * in a javascript map like format.
152      *
153      * @param cols JavaScript like map of color names
154      * @since 0.9
155      */
156     public void setColors(final String cols) {
157         for (final String item : cols.split(MulticolorLayout.SPLIT_ITEMS)) {
158             final String[] values = item.split(MulticolorLayout.SPLIT_VALUES);
159             this.colors.put(values[0], values[1]);
160         }
161         /**
162          * If setConversionPattern was called before me must call again
163          * to be sure to replace all custom color constants with
164          * new values.
165          */
166         if (this.base != null) {
167             this.setConversionPattern(this.base);
168         }
169     }
170 
171     /**
172      * Allow to overwrite the ANSI color values for the log levels
173      * in a javascript map like format.
174      *
175      * @param lev JavaScript like map of levels
176      * @since 0.9
177      */
178     public void setLevels(final String lev) {
179         for (final String item : lev.split(MulticolorLayout.SPLIT_ITEMS)) {
180             final String[] values = item.split(MulticolorLayout.SPLIT_VALUES);
181             final String level = values[0].toUpperCase(Locale.ENGLISH);
182             if (Level.toLevel(level, null) == null) {
183                 throw new IllegalArgumentException(
184                     String.format(Locale.ENGLISH, "Unknown level '%s'", level)
185                 );
186             }
187             this.levels.put(level, values[1]);
188         }
189     }
190 
191     @Override
192     public String format(final LoggingEvent event) {
193         return super.format(event).replace(
194             String.format("%s?m", MulticolorLayout.CSI),
195             String.format(
196                 "%s%sm",
197                 MulticolorLayout.CSI,
198                 this.levels.get(event.getLevel().toString())
199             )
200         );
201     }
202 
203     /**
204      * Convert our text to ANSI color.
205      * @param meta Meta text
206      * @return ANSI color
207      */
208     private String ansi(final String meta) {
209         final String ansi;
210         if (meta == null) {
211             ansi = "?";
212         } else if (meta.matches("[a-z]+")) {
213             ansi = this.colors.get(meta);
214             if (ansi == null) {
215                 throw new IllegalArgumentException(
216                     String.format("unknown color '%s'", meta)
217                 );
218             }
219         } else {
220             ansi = meta;
221         }
222         return ansi;
223     }
224 
225     /**
226      * Color map.
227      * @return Map of colors
228      */
229     private static ConcurrentMap<String, String> colorMap() {
230         final ConcurrentMap<String, String> map =
231             new ConcurrentHashMap<String, String>();
232         map.put("black", "30");
233         map.put("blue", "34");
234         map.put("cyan", "36");
235         map.put("green", "32");
236         map.put("magenta", "35");
237         map.put("red", "31");
238         map.put("yellow", "33");
239         map.put("white", "37");
240         return map;
241     }
242 
243     /**
244      * Level map.
245      * @return Map of levels
246      */
247     private static ConcurrentMap<String, String> levelMap() {
248         final ConcurrentMap<String, String> map =
249             new ConcurrentHashMap<String, String>();
250         map.put(Level.TRACE.toString(), "2;33");
251         map.put(Level.DEBUG.toString(), "2;37");
252         map.put(Level.INFO.toString(), "0;37");
253         map.put(Level.WARN.toString(), "0;33");
254         map.put(Level.ERROR.toString(), "0;31");
255         map.put(Level.FATAL.toString(), "0;35");
256         return map;
257     }
258 
259 }
